1. 特征数据的独热编码

独热编码

有时数据集中有类别类型的数据, 在进行分析时需要将这些数据转化为数值型, 这时候就需要对数据进行独热编码. 独热编码时需要将每一个元组(featureID, category)表示成为唯一的整数. 示例如下:

  1. # 样本1有两个属性, 样本2和样本3有三个属性
  2. sampleOne = [(0, 'mouse'), (1, 'black')]
  3. sampleTwo = [(0, 'cat'), (1, 'tabby'), (2, 'mouse')]
  4. sampleThree =  [(0, 'bear'), (1, 'black'), (2, 'salmon')]
  5. sampleDataRDD = sc.parallelize([sampleOne, sampleTwo, sampleThree])
  7. # 将每一个属性, 映射到一个整数, 整数编码从0开始
  8. # 比如说, 标识为0的有三种类型, 分别是'bear', 'cat', 和'mouse'
  9. # 需要依次赋值
  10. sampleOHEDictManual = {}
  11. sampleOHEDictManual[(0,'bear')] = 0
  12. sampleOHEDictManual[(0,'cat')] = 1
  13. sampleOHEDictManual[(0,'mouse')] = 2
  14. sampleOHEDictManual[(1, 'black')] = 3
  15. sampleOHEDictManual[(1, 'tabby')] = 4
  16. sampleOHEDictManual[(2, 'mouse')] = 5
  17. sampleOHEDictManual[(2, 'salmon')] = 6
  19. # 此时, 这个字典就是对类别数据的独热编码.

  1. # 相当于一条从0开始的射线上, 标注为2的次方的地方边界, 把次数划分为几块

logloss用来评估模型的优劣, 其计算公式如下:

  1. from math import log
  3. def computeLogLoss(p, y):
  4.     """Calculates the value of log loss for a given probabilty and label.
  6.     Note:
  7.         log(0) is undefined, so when p is 0 we need to add a small value (epsilon) to it
  8.         and when p is 1 we need to subtract a small value (epsilon) from it.
  10.     Args:
  11.         p (float): A probabilty between 0 and 1.
  12.         y (int): A label.  Takes on the values 0 and 1.
  14.     Returns:
  15.         float: The log loss value.
  16.     """
  17.     epsilon = 10e-12
  18.     if p == 0:
  19.         p += epsilon
  20.     elif p == 1:
  21.         p -= epsilon
  22.     if y == 1:
  23.         return -log(p)
  24.     else:
  25.         return -log(1 - p)

将预测值转为概率值

  1. from math import exp #  exp(-t) = e^-t
  3. def getP(x, w, intercept):
  4.     """Calculate the probability for an observation given a set of weights and intercept.
  6.     Note:
  7.         We'll bound our raw prediction between 20 and -20 for numerical purposes.
  9.     Args:
  10.         x (SparseVector): A vector with values of 1.0 for features that exist in this observation and 0.0 otherwise.
  11.         w (DenseVector): A vector of weights (betas) for the model.
  12.         intercept (float): The model's intercept.
  14.     Returns:
  15.         float: A probability between 0 and 1.
  16.     """
  17.     rawPrediction = w.dot(x) + intercept
  19.     # Bound the raw prediction value
  20.     rawPrediction = min(rawPrediction, 20)
  21.     rawPrediction = max(rawPrediction, -20)
  22.     return 1/(1 + exp(-rawPrediction))

5. 通过feature hashing来减少特征数据的维数

以上模型中, 通过独热编码将数据转化后, 向量长度较大, 在本文使用的数据中, 长度为233k, 实际的数据可能包含上千万个属性类别, 如果依然使用独热编码, 最终的向量将会非常大. 为了减少属性的维度, 我们需要使用feature hashing技术.

  1. from collections import defaultdict
  2. import hashlib
  4. # 返回结果中key表示桶号, value表示hash到这个桶的属性元组的个数
  5. def hashFunction(numBuckets, rawFeats, printMapping=False):
  6.     """Calculate a feature dictionary for an observation's features based on hashing.
  8.     Note:
  9.         Use printMapping=True for debug purposes and to better understand how the hashing works.
  11.     Args:
  12.         numBuckets (int): Number of buckets to use as features.
  13.         rawFeats (list of (int, str)): A list of features for an observation.  Represented as
  14.             (featureID, value) tuples.
  15.         printMapping (bool, optional): If true, the mappings of featureString to index will be
  16.             printed.
  18.     Returns:
  19.         dict of int to float:  The keys will be integers which represent the buckets that the
  20.             features have been hashed to.  The value for a given key will contain the count of the
  21.             (featureID, value) tuples that have hashed to that key.
  22.     """
  23.     mapping = {}
  24.     for ind, category in rawFeats:
  25.         featureString = category + str(ind)
  26.         mapping[featureString] = int(int(hashlib.md5(featureString).hexdigest(), 16) % numBuckets)
  27.     if(printMapping): print mapping
  28.     sparseFeatures = defaultdict(float)
  29.     for bucket in mapping.values():
  30.         sparseFeatures[bucket] += 1.0
  31.     return dict(sparseFeatures)
